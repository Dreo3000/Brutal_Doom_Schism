	ACTOR LungePunch : FastProjectile
	{
		Radius 10
		Height 5
		Speed 1
		Projectile
		+INVISIBLE
		+SKYEXPLODE
		States
		{
			Spawn:
				TNT1 A 0
				TNT1 A 1 A_ScaleVelocity(62) //Goes 62 units per tic.  This is to not go beyond the sudden death fall damage threshold in Hexen.
				TNT1 A 0 A_ChangeVelocity(velx, vely, .9*velz, CVF_REPLACE|AAPTR_PLAYER1)	//This changes the player's velocity.
				TNT1 AAAAAA 1 A_ChangeVelocity(velx, vely, .9*velz, CVF_REPLACE|AAPTR_PLAYER1)	//This keeps the player's velocity on track.
			Death:
				TNT1 AAA 1
				TNT1 A 0 A_GiveInventory("Hit", 1, AAPTR_TARGET)	//This is to prevent the player from using the flying punch immediately after and just flying around.
				TNT1 A 0 A_ChangeVelocity(0,0,0,CVF_REPLACE|AAPTR_PLAYER1)
				TNT1 A 1 A_ChangeVelocity(0,0,0,CVF_REPLACE)
				TNT1 A 0 A_ChangeVelocity(2,0,0,CVF_REPLACE | CVF_RELATIVE | AAPTR_TARGET)
							Stop
		}
	}
