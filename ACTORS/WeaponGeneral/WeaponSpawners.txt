ACTOR ShotgunSpawner: CustomInventory
{
 scale 0.8
 radius 24
 height 24
 +THRUACTORS +FORCEXYBILLBOARD
 Inventory.PickupMessage "$GOTSHOTGUN"
 Inventory.PickupSound "weapons/sgpump"
 Mass 1
    States
    {
	 Spawn:
		TNT1 A 0
		TNT1 A 2
		TNT1 A 0 A_ChangeFlag("THRUACTORS", 0)
	Stay:
		SHTC A 1
		SHTC A -1
		Stop
	Pickup:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("Shot_Gun",1,"GiveAmmo") //player already have shotgun - give ammo and goodbye
		TNT1 A 0 A_JumpIfInventory("SSG",1,"HaveSSG")		//Case 1
		TNT1 A 0 A_JumpIfInventory("AssaultShotgun",1,"HaveAssaultShotgun") //Case 2
		TNT1 A 0 A_JumpIfInventory("QuadShotgun",1,"HaveQuadShotgun") //Case 3
		TNT1 A 0 A_GiveInventory("Shot_Gun",1)  			//If player have not any of this, then his slot is empty - give shotgun and ammo
		Goto GiveAmmo
				
		HaveSSG: //player have SSG (1 of 4 weapons in slot)
		TNT1 A 0 A_JumpIfInventory("AssaultShotgun",1,"GiveAmmo") 	//if player have 2 of 4 weapons (full slot) then give only ammo 
		TNT1 A 0 A_JumpIfInventory("QuadShotgun",1,"GiveAmmo") 		//if player have another second weapon then give only ammo
		TNT1 A 0 A_GiveInventory("Shot_Gun",1)	//If player have only SSG, but not AS or QSSG, then give shotgun and ammo
		goto GiveAmmo
	
		HaveAssaultShotgun: //player have AssaultShotgun (1 of 4 weapons in slot)
		TNT1 A 0 A_JumpIfInventory("SSG",1,"GiveAmmo") 	//if player have 2 of 4 weapons (full slot) then give only ammo 
		TNT1 A 0 A_JumpIfInventory("QuadShotgun",1,"GiveAmmo") 		//if player have another second weapon then give only ammo
		TNT1 A 0 A_GiveInventory("Shot_Gun",1)	//If player have only Assault Shotugn, but not SSG or QSSG, then give shotgun and ammo
		goto GiveAmmo	
	
		HaveQuadShotgun:
		TNT1 A 0 A_JumpIfInventory("SSG",1,"GiveAmmo") 	//if player have 2 of 4 weapons (full slot) then give only ammo 
		TNT1 A 0 A_JumpIfInventory("AssaultShotgun",1,"GiveAmmo") 		//if player have another second weapon then give only ammo
		TNT1 A 0 A_GiveInventory("Shot_Gun",1)	//If player have only Quad Shotugn, but not SSG or Assault Shotgun, then give shotgun and ammo
		goto GiveAmmo		

	GiveAmmo:
		TNT1 A 0 A_GiveInventory("AmmoShell", 8)
		Stop
    }
}
